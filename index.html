function calculate(a, b, operation) { return new Promise((resolve, reject) => { if (typeof a !== 'number' || typeof b !== 'number') { reject('Invalid input. Both inputs must be numbers.'); return; } switch (operation) { case '+': resolve(a + b); break; case '-': resolve(a - b); break; case '*': resolve(a * b); break; case '/': if (b === 0) { reject('Division by zero is not allowed.'); } else { resolve(a / b); } break; default: reject('Invalid operation. Supported operations are +, -, *, and /.'); } }); } // Usage examples: calculate(5, 3, '+') .then(result => console.log(`Result: ${result}`)) .catch(error => console.error(`Error: ${error}`)); calculate(10, 2, '/') .then(result => console.log(`Result: ${result}`)) .catch(error => console.error(`Error: ${error}`)); calculate(8, 0, '/') .then(result => console.log(`Result: ${result}`)) .catch(error => console.error(`Error: ${error}`)); calculate(7, 'abc', '*') .then(result => console.log(`Result: ${result}`)) .catch(error => console.error(`Error: ${error}`)); calculate(9, 3, '^') .then(result => console.log(`Result: ${result}`)) .catch(error => console.error(`Error: ${error}`));
